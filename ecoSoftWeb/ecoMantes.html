			@DESCRIPCION varchar(200),

			@VISIBLE bit,
			@HIJO bit,
			@SEPARADOR bit,
			@TEXTO_ALTERNATIVO varchar(200),
			@MANTE_DESTIN int,
			@TIPO bit,
			@BBDD varchar(60),
			@USUARIO_SQL varchar(50),	          
			@SISTEMA_OPERATIVO VARCHAR(200),
			@NAVEGADOR VARCHAR(200),
			@VERSION_NAVEGADOR VARCHAR(50),
			@IP_LOCAL VARCHAR(27),In
